class Solution {
   
    public static boolean checkPangram(String s) {
        
        HashSet<Character> charSet = new HashSet<>();
        for (int i=0; i < s.length(); i++) {
            char ch = Character.toLowerCase(s.charAt(i));

            if (Character.isLetter(ch)) {
                charSet.add(ch);
            }
        }
        return charSet.size() == 26;
    }
}
